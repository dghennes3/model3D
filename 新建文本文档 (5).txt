            function startDrag(e) {
                isDragging = true;
                lastX = e.clientX || e.pageX;
                lastY = e.clientY || e.pageY;
                lastTime = Date.now();
            }

            function drag(e) {
                if (!isDragging) return;
                const currentX = e.clientX || e.pageX;
                const currentY = e.clientY || e.pageY;
                const currentTime = Date.now();
                
                const deltaX = currentX - lastX;
                const deltaY = currentY - lastY;
                const timeDiff = Math.max(1, currentTime - lastTime);
                const speedX = Math.abs(deltaX) / timeDiff;
                const speedY = Math.abs(deltaY) / timeDiff;
                
                const powerFactorX = 0.1 + (speedX * 0.5);
                const powerFactorY = 0.1 + (speedY * 0.5);
                const changeX = deltaX * powerFactorX;
                const changeY = deltaY * powerFactorY;
                
                // 根据当前器型限制参数范围
                switch(selectedShape) {
                    case 'vase':
                        currentParams.topRadiusX = Math.max(20, Math.min(70, currentParams.topRadiusX + changeX));
                        currentParams.bottomRadiusX = Math.max(40, Math.min(90, currentParams.bottomRadiusX + changeX * 0.8));
                        currentParams.topRadiusY = Math.max(10, Math.min(50, currentParams.topRadiusY + changeY));
                        currentParams.bottomRadiusY = Math.max(60, Math.min(95, currentParams.bottomRadiusY + changeY * 0.8));
                        break;
                    case 'bowl':
                        currentParams.topRadiusX = Math.max(30, Math.min(70, currentParams.topRadiusX + changeX));
                        currentParams.bottomRadiusX = Math.max(10, Math.min(40, currentParams.bottomRadiusX + changeX * 0.8));
                        currentParams.topRadiusY = Math.max(20, Math.min(60, currentParams.topRadiusY + changeY));
                        currentParams.bottomRadiusY = Math.max(5, Math.min(20, currentParams.bottomRadiusY + changeY * 0.8));

                    case 'plate':

                const dirLight2 = new THREE.DirectionalLight(0xffffff, 0.4);
                dirLight2.position.set(-5, -5, 5);
                modelViewerScene.add(dirLight2);
                
                const hemisphereLight = new THREE.HemisphereLight(0xffffff, 0x444444);
                hemisphereLight.position.set(0, 20, 0);
                modelViewerScene.add(hemisphereLight);
                
                modelViewerCamera = new THREE.PerspectiveCamera(75, container.clientWidth / container.clientHeight, 0.1, 1000);
                modelViewerCamera.position.z = 8;
                
                modelViewerRenderer = new THREE.WebGLRenderer({ antialias: true });
                modelViewerRenderer.setSize(container.clientWidth, container.clientHeight);
                modelViewerRenderer.toneMapping = THREE.ACESFilmicToneMapping;
                modelViewerRenderer.toneMappingExposure = 1.0;
                container.appendChild(modelViewerRenderer.domElement);
                
                modelViewerControls = new THREE.OrbitControls(modelViewerCamera, modelViewerRenderer.domElement);
                modelViewerControls.enableDamping = true;
                modelViewerControls.dampingFactor = 0.05;
                modelViewerControls.enablePan = true;
                modelViewerControls.enableZoom = true;
                
                // 窗口大小适配
                window.addEventListener('resize', () => {
                    modelViewerCamera.aspect = container.clientWidth / container.clientHeight;
                    modelViewerCamera.updateProjectionMatrix();
                    modelViewerRenderer.setSize(container.clientWidth, container.clientHeight);
                });
                
                // 动画循环
                function animate() {
                    requestAnimationFrame(animate);
                    modelViewerControls.update();
                    modelViewerRenderer.render(modelViewerScene, modelViewerCamera);
                }
                animate();
            }
            
            // 根据朝代加载不同模型
            loadPeriodModel(period);
        }
